CREATE TABLE movie
(
id int NOT NULL AUTO_INCREMENT,
director varchar(255),
title varchar(255),
language varchar(255),
doubanId varchar(255),
country varchar(255),
summary varchar(255),
flash varchar(255),
poster varchar(255),
year int,
coutry varchar(255),
pv int NOT NULL DEFAULT 0,
category int,
createAt datetime,
updateAt datetime,
genres varchar(255),
PRIMARY KEY (id)
)


CREATE TABLE category
(
id int NOT NULL AUTO_INCREMENT,
name varchar(255),
movieId int,
createAt datetime,
updateAt datetime,
PRIMARY KEY (id),
FOREIGN KEY (movieId) REFERENCES movie(id)
)


CREATE TABLE users
(
name varchar(255),
password varchar(255),
role int NOT NULL DEFAULT 0,
createAt datetime,
updateAt datetime,
UNIQUE (name)
)



CREATE TABLE users
(
Id_P int NOT NULL,

FirstName varchar(255),

UNIQUE (Id_P)
